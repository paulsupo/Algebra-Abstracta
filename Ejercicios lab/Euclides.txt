ZZ positivo(ZZ a){
  if(a<0)
    a*=-1;
  return a;
}

ZZ modulo(ZZ a, ZZ b){
    if(a>=0)
        return a-(a/b)*b;
    else
        return a-((a/b)-1)*b;
}


//Execution Time: 8e-06
ZZ algoritmo1(ZZ a,ZZ b){
  //cout<<"Ingreso con los numeros "<<a<<" y "<<b<<endl;
  ZZ r;
  while(true){
    r=modulo(a,b);
    //cout<<"r= "<<r<<endl;
    if(r==0)
      return b;
    a=b;
    //cout<<"a= "<<a<<endl;
    b=r;
    //cout<<"b= "<<b<<endl<<endl;
  }
}
